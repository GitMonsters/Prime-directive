Quick Installation

Get started in under a minute

# Download the latest release
curl -LO https://github.com/GitMonsters/Prime-directive/releases/latest/download/rustyworm-linux-amd64

# Make it executable
chmod +x rustyworm-linux-amd64

# Move to PATH
sudo mv rustyworm-linux-amd64 /usr/local/bin/rustyworm

# Verify installation
rustyworm --version
# Pull the image
docker pull ghcr.io/gitmonsters/rustyworm:latest

# Run with Docker Compose
curl -LO https://raw.githubusercontent.com/GitMonsters/Prime-directive/main/deploy/docker-compose.yml
docker compose up -d

# Check status
docker compose ps
# Clone the repository
git clone https://github.com/GitMonsters/Prime-directive.git
cd Prime-directive

# Build with all features
cargo build --release --features full

# Run tests
cargo test --features full --lib

# Install
sudo cp target/release/rustyworm /usr/local/bin/
